class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        d = {
            "(": ")",
            "{": "}",
            "[": "]"
        }
        for item in s:
            if item in d:
                stack.append(d[item])
            elif not stack or stack.pop() != item:
                return False
        
        return len(stack) == 0

时间复杂度：O(n)
